
Explain the output of the following C++ program:
```cpp
#include <iostream>
using namespace std;

int main() {
    int *pa = new int[5];
    int *pm = pa+2;
    *(pm-1) = *(pm+1) = 1;
    *pa = 3;
    *pm = *pa+1;
    pa[4] = 5;

    for (int i = 0; i < 5; ++i)
        cout << pa[i] << " ";
    cout << endl;

    delete[] pa;

    return 0;
}
```
